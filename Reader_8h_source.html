270 virtual void load() final override {
271 // determine the length we want to request depending
272 // on the number of entries left in the file
273 std::size_t request_len = this->max_len_;
274 if (request_len + i_file_ > entries_) {
275 request_len = entries_ - i_file_;
276 assert(request_len >= 0);
277 }
278 // load the next chunk into memory
279 if constexpr (std::is_same_v<AtomicType,bool>) {
280 // get around std::vector<bool> specialization
281 auto buff = std::make_unique<bool[]>(request_len);
282 this->set_.select({i_file_}, {request_len})
283 .read(buff.get(), HighFive::AtomicType<bool>());
284 buffer_.reserve(request_len);
285 for (std::size_t i{0}; i < request_len; i++)
286 buffer_.push_back(buff[i]);
287 } else {
288 this->set_.select({i_file_}, {request_len}).read(buffer_);
289 }
290 // update indices
291 i_file_ += buffer_.size();
292 i_memory_ = 0;
293 }
294 };
