191 virtual ~Buffer() = default;
200 void save(const AtomicType& val) {
201 buffer_.push_back(val);
202 if (buffer_.size() > this->max_len_) flush();
203 }
230 virtual void flush() final override {
231 if (buffer_.size() == 0) return;
232 std::size_t new_extent = i_file_ + buffer_.size();
233 // throws if not created yet
234 if (this->set_.getDimensions().at(0) < new_extent) {
235 this->set_.resize({new_extent});
236 }
237 if constexpr (std::is_same_v<AtomicType, bool>) {
238 // handle bool specialization
239 auto buff = std::make_unique<bool[]>(buffer_.size());
240 for (std::size_t i{0}; i < buffer_.size(); i++)
241 buff[i] = buffer_.at(i);
242 this->set_.select({i_file_}, {buffer_.size()})
243 .write_raw(buff.get(), HighFive::AtomicType<bool>());
244 } else {
245 this->set_.select({i_file_}, {buffer_.size()}).write(buffer_);
246 }
247 i_file_ += buffer_.size();
248 buffer_.clear();
249 buffer_.reserve(this->max_len_);
250 }
