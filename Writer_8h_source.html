84 void save(const std::string& path, const AtomicType& val) {
85 static_assert(
86 is_atomic_v<AtomicType>,
87 "Type unsupported by HighFive as Atomic made its way to Writer::save");
88 if (buffers_.find(path) == buffers_.end()) {
89 // first save attempt, need to create the data set to be saved
90 // - we pass the newly created dataset to the buffer to hold onto
91 // for flushing purposes
92 // - the length of the buffer is the same size as the chunks in
93 // HDF5, this is done on purpose
94 // - if the type is a bool, we define the HighFive type to be
95 // our custom enum which mimics the type used by h5py
96 HighFive::DataType t;
97 if constexpr (std::is_same_v<AtomicType,bool>) {
98 t = create_enum_bool();
99 } else {
100 t = HighFive::AtomicType<AtomicType>();
101 }
102 buffers_.emplace(
103 path, std::make_unique<Buffer<AtomicType>>(
104 rows_per_chunk_,
105 file_.createDataSet(path, space_,t,create_props_)));
106 }
107 dynamic_cast<Buffer<AtomicType>&>(*buffers_.at(path)).save(val);
108 }

192 explicit Buffer(std::size_t max, HighFive::DataSet s)
193 : BufferHandle(max, s), buffer_{}, i_file_{0} {
194 buffer_.reserve(this->max_len_);
195 }
197 virtual ~Buffer() = default;
206 void save(const AtomicType& val) {
207 buffer_.push_back(val);
208 if (buffer_.size() > this->max_len_) flush();
209 }
239 virtual void flush() final override {
240 if (buffer_.size() == 0) return;
241 std::size_t new_extent = i_file_ + buffer_.size();
242 // throws if not created yet
243 if (this->set_.getDimensions().at(0) < new_extent) {
244 this->set_.resize({new_extent});
245 }
246 if constexpr (std::is_same_v<AtomicType, bool>) {
247 // handle bool specialization
248 std::vector<Bool> buff;
249 buff.reserve(buffer_.size());
250 for (const auto& v : buffer_) buff.push_back(v ? Bool::TRUE : Bool::FALSE);
251 this->set_.select({i_file_}, {buffer_.size()}).write(buff);
252 } else {
253 this->set_.select({i_file_}, {buffer_.size()}).write(buffer_);
254 }
255 i_file_ += buffer_.size();
256 buffer_.clear();
257 buffer_.reserve(this->max_len_);
258 }
