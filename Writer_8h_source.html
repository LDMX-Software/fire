99
100 public:
101 explicit Buffer(std::size_t max, HighFive::DataSet s)
102 : BufferHandle(max, s), buffer_{}, i_file_{0} {
103 buffer_.reserve(this->max_len_);
104 }
105 virtual ~Buffer() = default;
106 void save(const AtomicType& val) {
107 buffer_.push_back(val);
108 if (buffer_.size() > this->max_len_) flush();
109 }
110 virtual void flush() final override {
111 if (buffer_.size() == 0) return;
112 std::size_t new_extent = i_file_ + buffer_.size();
113 // throws if not created yet
114 if (this->set_.getDimensions().at(0) < new_extent) {
115 this->set_.resize({new_extent});
116 }
117 if constexpr (std::is_same_v<AtomicType, bool>) {
118 // handle bool specialization
119 std::vector<short> buff;
120 buff.reserve(buffer_.size());
121 for (const auto& v : buffer_) buff.push_back(v);
122 this->set_.select({i_file_}, {buff.size()}).write(buff);
123 } else {
124 this->set_.select({i_file_}, {buffer_.size()}).write(buffer_);
125 }
126 i_file_ += buffer_.size();
127 buffer_.clear();
128 buffer_.reserve(this->max_len_);
129 }
130 };
