185 template <typename MemberType>
186 void attach(const std::string& name, MemberType& m, SaveLoad sl = SaveLoad::Both) {
187 if (name == constants::SIZE_NAME) {
188 throw Exception("BadName",
189 "The member name '"+constants::SIZE_NAME+"' is not allowed due to "
190 "its use in the serialization of variable length types.\n"
191 " Please give your member a more detailed name corresponding to "
192 "your class", false);
193 }
194 bool save{false}, load{false};
195 Reader* input_file{input_file_};
196 if (sl == SaveLoad::LoadOnly) load = true;
197 else if (sl == SaveLoad::SaveOnly) { save = true; input_file = nullptr; }
198 else { save = true; load = true; }
199 members_.push_back(std::make_tuple(save, load,
200 std::make_unique<Data<MemberType>>(this->path_ + "/" + name, input_file, &m)));
201 }
