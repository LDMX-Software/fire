309 virtual void load() final override {
310 // determine the length we want to request depending
311 // on the number of entries left in the file
312 std::size_t request_len = this->max_len_;
313 if (request_len + i_file_ > entries_) {
314 request_len = entries_ - i_file_;
315 assert(request_len >= 0);
316 }
317 // load the next chunk into memory
318 if constexpr (std::is_same_v<AtomicType,bool>) {
324 std::vector<Bool> buff;
325 buff.resize(request_len);
326 this->set_.select({i_file_}, {request_len})
327 .read(buff.data(),create_enum_bool());
328 buffer_.reserve(buff.size());
329 for (const auto& v : buff) buffer_.push_back(v == Bool::TRUE);
330 } else {
331 this->set_.select({i_file_}, {request_len}).read(buffer_);
332 }
333 // update indices
334 i_file_ += buffer_.size();
335 i_memory_ = 0;
336 }
