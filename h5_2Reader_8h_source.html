280 virtual void load() final override {
281 // determine the length we want to request depending
282 // on the number of entries left in the file
283 std::size_t request_len = this->max_len_;
284 if (request_len + i_file_ > entries_) {
285 request_len = entries_ - i_file_;
286 assert(request_len >= 0);
287 }
288 // load the next chunk into memory
289 if constexpr (std::is_same_v<AtomicType,bool>) {
290 // get around std::vector<bool> specialization
291 auto buff = std::make_unique<bool[]>(request_len);
292 this->set_.select({i_file_}, {request_len})
293 .read(buff.get(), HighFive::AtomicType<bool>());
294 buffer_.reserve(request_len);
295 for (std::size_t i{0}; i < request_len; i++)
296 buffer_.push_back(buff[i]);
297 } else {
298 this->set_.select({i_file_}, {request_len}).read(buffer_);
299 }
300 // update indices
301 i_file_ += buffer_.size();
302 i_memory_ = 0;
303 }
304 };
