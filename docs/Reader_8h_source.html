124  // get the number of entries for later checking
125  entries_ = this->set_.getDimensions().at(0);
126  // do first load upon creation
127  this->load();

129  virtual ~Buffer() = default;
130  void read(AtomicType& out) {
131  if (i_memory_ == buffer_.size()) this->load();
132  out = buffer_[i_memory_];
133  i_memory_++;
134  }
135  virtual void load() final override {
136  // determine the length we want to request depending
137  // on the number of entries left in the file
138  std::size_t request_len = this->max_len_;
139  if (request_len + i_file_ > entries_) {
140  request_len = entries_ - i_file_;
141  }
142  // load the next chunk into memory
143  if constexpr (std::is_same_v<AtomicType,bool>) {
144  // get around std::vector<bool> specialization
145  std::vector<short> buff;
146  this->set_.select({i_file_}, {request_len}).read(buff);
147  buffer_.reserve(buff.size());
148  for (const auto& v : buff) buffer_.push_back(v);
149  } else {
150  this->set_.select({i_file_}, {request_len}).read(buffer_);
151  }
152  // update indices
153  i_file_ += buffer_.size();
154  i_memory_ = 0;
155  }
