191 template <typename DataType>
192 void add(const std::string& name, const DataType& data) {
193 static const bool ADD_KEEP_DEFAULT = true;
194 std::string full_name{fullName(name, pass_)};
195 if (objects_.find(full_name) == objects_.end()) {
196 // check available_objects_ listing so we don't in-advertently replace
197 // any datasets of the same name read in from the inputfile
198 // we know we are worried about data from the input file
199 // because data from previous producers in the sequence
200 // would already exist in the objects_ map
201 // we rely on trusting that setInputFile gets the listing
202 // of event objects from the input file and puts them
203 // into availble_objects_
204 if (search("^"+name+"$","^"+pass_+"$",".*").size() > 0) {
205 throw Exception("Repeat",
206 "Data named "+full_name+" already exists in the input file.");
207 }
208 available_objects_.emplace_back(name, pass_,
209 boost::core::demangle(typeid(DataType).name()));
210
211 // a data set hasn't been created for this data yet
212 // we good, lets create the new data set
213 //
214 // NOTES:
215 // - check if new data is going to be written to output file
216 // or just used during this run
217 // - without any applicable drop/keep rules,
218 // we do save these datasets
219 // - we mark these objects as should_load == false because
220 // they are new and not from an input file
221 auto& obj{objects_[full_name]};
222 obj.data_ = std::make_unique<h5::Data<DataType>>(h5::constants::EVENT_GROUP+"/"+full_name);
223 obj.should_save_ = keep(full_name, ADD_KEEP_DEFAULT);
224 obj.should_load_ = false;
225 obj.updated_ = false;
226
227 // if we are saving this object, we should save the default value for all entries
228 // up to this one. This (along with 'clearing' at the end of each event)
229 // allows users to asyncronously add event objects and the events without an 'add'
230 // have a 'default' or 'cleared' object value.
231 if (obj.should_save_) {
232 obj.data_->clear();
233 for (std::size_t i{0}; i < i_entry_; i++)
234 obj.data_->save(output_file_);
235 }
236 }
237
238 auto& obj{objects_.at(full_name)};
239 if (obj.updated_) {
240 // this data set has been updated by another processor in the sequence
241 throw Exception("Repeat",
242 "Data named " + full_name + " already added to the event"
243 " by a previous producer in the sequence.");
244 }
245
246 try {
248 obj.getDataRef<DataType>().update(data);
249 obj.updated_ = true;
250 } catch (std::bad_cast const&) {
251 throw Exception("TypeMismatch",
252 "Data corresponding to " + full_name + " has different type.");
253 }
254 }

294 template <typename DataType>
295 const DataType& get(const std::string& name,
296 const std::string& pass = "") const {
297 static const bool GET_KEEP_DEFAULT = false;
298 std::string full_name;
299 if (not pass.empty()) {
300 // easy case, pass was specified explicitly
301 full_name = fullName(name, pass);
302 } else if (known_lookups_.find(name) != known_lookups_.end()) {
303 full_name = known_lookups_.at(name);
304 } else {
305 // need to search current (and potential) available_objects using partial name
306 auto type = boost::core::demangle(typeid(DataType).name());
307 auto options{search("^" + name + "$", ".*", "^" + type + "$")};
308 if (options.size() == 0) {
309 throw Exception("Miss",
310 "Data " + name + " of type " + type + " not found.");
311 } else if (options.size() > 1) {
312 throw Exception("Ambig",
313 "Data " + name + " of type " + type + " is ambiguous. Provide a pass name.");
314 }
315
316 // exactly one option
317 full_name = fullName(options.at(0).name(), options.at(0).pass());
318 // add into cache
319 known_lookups_[name] = full_name;
320 }
321
322 if (objects_.find(full_name) == objects_.end()) {
323 // final check for input file, never should enter here without one
324 if (not input_file_) {
325 throw Exception("Miss",
326 "Data " + full_name + " was not created by an earlier processor "
327 "and there is not input file to attempt to read it from.");
328 }
329 // a data set hasn't been created for this data yet
330 // we good, lets create the new data set
331 //
332 // NOTES:
333 // - check if new data is going to be written to output file
334 // or just used during this run
335 // - without any applicable drop/keep rules,
336 // we do save these datasets
337 // - we mark these objects as should_load == false because
338 // they are new and not from an input file
339 auto& obj{objects_[full_name]};
340 obj.data_ = std::make_unique<h5::Data<DataType>>(h5::constants::EVENT_GROUP+"/"+full_name);
341 obj.should_save_ = keep(full_name, GET_KEEP_DEFAULT);
342 obj.should_load_ = true;
343 obj.updated_ = false;
344 // get this object up to the current entry
345 // loading may throw an H5 error if DataType doesn't match the type
346 // being read from the input file
347 for (std::size_t i{0}; i < i_entry_+1; i++)
348 obj.data_->load(*input_file_);
349 }
350
351 // type casting, 'bad_cast' thrown if unable
352 try {
353 return objects_[full_name].getDataRef<DataType>().get();
354 } catch (const std::bad_cast&) {
355 throw Exception("BadType",
356 "Data corresponding to " + full_name + " has different type.");
357 }
