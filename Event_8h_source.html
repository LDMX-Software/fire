203 template <typename DataType>
204 void add(const std::string& name, const DataType& data) {
205 static const bool ADD_KEEP_DEFAULT = true;
206 std::string full_name{fullName(name, pass_)};
207 if (objects_.find(full_name) == objects_.end()) {
208 // check available_objects_ listing so we don't in-advertently replace
209 // any datasets of the same name read in from the inputfile
210 // we know we are worried about data from the input file
211 // because data from previous producers in the sequence
212 // would already exist in the objects_ map
213 // we rely on trusting that setInputFile gets the listing
214 // of event objects from the input file and puts them
215 // into availble_objects_
216 if (search("^"+name+"$","^"+pass_+"$",".*").size() > 0) {
217 throw Exception("Repeat",
218 "Data named "+full_name+" already exists in the input file.");
219 }
220 available_objects_.emplace_back(name, pass_,
221 boost::core::demangle(typeid(DataType).name()));
222
223 // a data set hasn't been created for this data yet
224 // we good, lets create the new data set
225 //
226 // NOTES:
227 // - check if new data is going to be written to output file
228 // or just used during this run
229 // - without any applicable drop/keep rules,
230 // we do save these datasets
231 // - we mark these objects as should_load == false because
232 // they are new and not from an input file
233 auto& obj{objects_[full_name]};
234 obj.data_ = std::make_unique<io::Data<DataType>>(io::constants::EVENT_GROUP+"/"+full_name);
235 obj.should_save_ = keep(full_name, ADD_KEEP_DEFAULT);
236 obj.should_load_ = false;
237 obj.updated_ = false;
238
239 // if we are saving this object, we should save the default value for all entries
240 // up to this one. This (along with 'clearing' at the end of each event)
241 // allows users to asyncronously add event objects and the events without an 'add'
242 // have a 'default' or 'cleared' object value.
243 if (obj.should_save_) {
244 obj.data_->clear();
245 for (std::size_t i{0}; i < i_entry_; i++)
246 obj.data_->save(output_file_);
247 }
248 }
249
250 auto& obj{objects_.at(full_name)};
251 if (obj.updated_) {
252 // this data set has been updated by another processor in the sequence
253 throw Exception("Repeat",
254 "Data named " + full_name + " already added to the event"
255 " by a previous producer in the sequence.");
256 }
257
258 try {
260 obj.getDataRef<DataType>().update(data);
261 obj.updated_ = true;
262 } catch (std::bad_cast const&) {
263 throw Exception("TypeMismatch",
264 "Data corresponding to " + full_name + " has different type.");
265 }
266 }

306 template <typename DataType>
307 const DataType& get(const std::string& name,
308 const std::string& pass = "") const {
309 static const bool GET_KEEP_DEFAULT = false;
310 std::string full_name, type;
311 if (not pass.empty()) {
312 // easy case, pass was specified explicitly
313 full_name = fullName(name, pass);
314 } else if (known_lookups_.find(name) != known_lookups_.end()) {
315 full_name = known_lookups_.at(name);
316 } else {
317 // need to search current (and potential) available_objects using partial name
318 auto options{search("^" + name + "$", ".*", ".*")};
319 if (options.size() == 0) {
320 throw Exception("Miss",
321 "Data " + name + " not found.");
322 } else if (options.size() > 1) {
323 throw Exception("Ambig",
324 "Data " + name + " is ambiguous. Provide a pass name.");
325 }
326
327 // exactly one option
328 full_name = fullName(options.at(0).name(), options.at(0).pass());
329 type = options.at(0).type();
330 // add into cache
331 known_lookups_[name] = full_name;
332 }
333
334 if (objects_.find(full_name) == objects_.end()) {
335 // final check for input file, never should enter here without one
336 if (not input_file_) {
337 throw Exception("Miss",
338 "Data " + full_name + " was not created by an earlier processor "
339 "and there is not input file to attempt to read it from.");
340 }
341 // a data set hasn't been created for this data yet
342 // we good, lets create the new data set
343 //
344 // NOTES:
345 // - check if new data is going to be written to output file
346 // or just used during this run
347 // - without any applicable drop/keep rules,
348 // we do save these datasets
349 // - we mark these objects as should_load == false because
350 // they are new and not from an input file
351 auto& obj{objects_[full_name]};
352 obj.data_ = std::make_unique<io::Data<DataType>>(io::constants::EVENT_GROUP+"/"+full_name);
353 obj.should_save_ = keep(full_name, GET_KEEP_DEFAULT);
354 obj.should_load_ = true;
355 obj.updated_ = false;
356 // get this object up to the current entry
357 // loading may throw an H5 error if the shape of the data on disk
358 // cannot be loaded into the input type
359 try {
360 for (std::size_t i{0}; i < i_entry_+1; i++) input_file_->load_into(*obj.data_);
361 } catch (const HighFive::DataSetException&) {
362 throw Exception("BadType",
363 "Data " + full_name + " could not be loaded into "
364 + boost::core::demangle(typeid(DataType).name())
365 + " from the type it was written as " + type);
366 }
367 }
368
369 // type casting, 'bad_cast' thrown if unable
370 try {
371 return objects_[full_name].getDataRef<DataType>().get();
372 } catch (const std::bad_cast&) {
373 throw Exception("BadType",
374 "Data " + full_name + " was initialy loaded with type " + type
375 + " which cannot be casted into "
376 + boost::core::demangle(typeid(DataType).name()));
377 }
378 }
