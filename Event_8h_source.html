124 template <typename DataType>
125 const DataType& get(const std::string& name,
126 const std::string& pass = "") const {
127 std::string full_name;
128 if (not pass.empty()) {
129 // easy case, pass was specified explicitly
130 full_name = fullName(name, pass);
131 } else if (known_lookups_.find(name) != known_lookups_.end()) {
132 full_name = known_lookups_.at(name);
133 } else {
134 // need to search current (and potential) products using partial name
135 auto type = boost::core::demangle(typeid(DataType).name());
136 auto options{search("^" + name + "$", "", "^" + type + "$")};
137 if (options.size() == 0) {
138 throw Exception("SetMiss",
139 "DataSet " + name + " of type " + type + " not found.");
140 } else if (options.size() > 1) {
141 throw Exception("SetAmbig",
142 "DataSet " + name + " of type " + type + " is ambiguous. Provide a pass name.");
143 }
144
145 // exactly one option
146 full_name = fullName(options.at(0).name(), options.at(0).pass());
147 // add into cache
148 known_lookups_[name] = full_name;
149 }
150
151 if (objects_.find(full_name) == objects_.end()) {
152 // final check for input file, never should enter here without one
153 assert(input_file_);
154 // a data set hasn't been created for this data yet
155 // we good, lets create the new data set
156 //
157 // NOTES:
158 // - check if new data is going to be written to output file
159 // or just used during this run
160 // - without any applicable drop/keep rules,
161 // we do save these datasets
162 // - we mark these objects as should_load == false because
163 // they are new and not from an input file
164 auto& obj{objects_[full_name]};
165 obj.set_ = std::make_unique<h5::DataSet<DataType>>(h5::Reader::EVENT_GROUP+"/"+full_name);
166 obj.should_save_ = keep(full_name, false);
167 obj.should_load_ = true;
168 // get this object up to the current entry
169 // loading may throw an H5 error
170 for (std::size_t i{0}; i < i_entry_+1; i++)
171 obj.set_->load(*input_file_);
172 }
173
174 // type casting, 'bad_cast' thrown if unable
175 try {
176 return objects_[full_name].getDataSetRef<DataType>().get();
177 } catch (const std::bad_cast&) {
178 throw Exception("BadType",
179 "DataSet corresponding to " + full_name + " has different type.");
180 }
