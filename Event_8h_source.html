240 template <typename DataType>
241 void add(const std::string& name, const DataType& data) {
242 static const bool ADD_KEEP_DEFAULT = true;
243 std::string full_name{fullName(name, pass_)};
244 if (objects_.find(full_name) == objects_.end()) {
245 // check available_objects_ listing so we don't in-advertently replace
246 // any datasets of the same name read in from the inputfile
247 // we know we are worried about data from the input file
248 // because data from previous producers in the sequence
249 // would already exist in the objects_ map
250 // we rely on trusting that setInputFile gets the listing
251 // of event objects from the input file and puts them
252 // into availble_objects_
253 if (search("^"+name+"$","^"+pass_+"$",".*").size() > 0) {
254 throw Exception("Repeat",
255 "Data named "+full_name+" already exists in the input file.");
256 }
257 auto& tag{available_objects_.emplace_back(name, pass_,
258 boost::core::demangle(typeid(DataType).name()),
259 keep(full_name, ADD_KEEP_DEFAULT))};
260 tag.loaded_ = true;
261
262 // a data set hasn't been created for this data yet
263 // we good, lets create the new data set
264 //
265 // NOTES:
266 // - check if new data is going to be written to output file
267 // or just used during this run
268 // - without any applicable drop/keep rules,
269 // we do save these datasets
270 // - we mark these objects as should_load == false because
271 // they are new and not from an input file
272 auto& obj{objects_[full_name]};
273 obj.data_ = std::make_unique<io::Data<DataType>>(io::constants::EVENT_GROUP+"/"+full_name);
274 obj.should_save_ = tag.keep_;
275 obj.should_load_ = false;
276 obj.updated_ = false;
277
278 // if we are saving this object, we should save the default value for all entries
279 // up to this one. This (along with 'clearing' at the end of each event)
280 // allows users to asyncronously add event objects and the events without an 'add'
281 // have a 'default' or 'cleared' object value.
282 if (obj.should_save_) {
283 obj.data_->clear();
284 for (std::size_t i{0}; i < i_entry_; i++)
285 obj.data_->save(output_file_);
286 }
287 }
288
289 auto& obj{objects_.at(full_name)};
290 if (obj.updated_) {
291 // this data set has been updated by another processor in the sequence
292 throw Exception("Repeat",
293 "Data named " + full_name + " already added to the event"
294 " by a previous producer in the sequence.");
295 }
296
297 try {
299 obj.getDataRef<DataType>().update(data);
300 obj.updated_ = true;
301 } catch (std::bad_cast const&) {
302 throw Exception("TypeMismatch",
303 "Data corresponding to " + full_name + " has different type.");
304 }
305 }

345 template <typename DataType>
346 const DataType& get(const std::string& name,
347 const std::string& pass = "") const {
348 std::string full_name, type;
349 if (not pass.empty()) {
350 // easy case, pass was specified explicitly
351 full_name = fullName(name, pass);
352 } else if (known_lookups_.find(name) != known_lookups_.end()) {
353 full_name = known_lookups_.at(name);
354 } else {
355 // need to search current (and potential) available_objects using partial name
356 auto options{search("^" + name + "$", ".*", ".*")};
357 if (options.size() == 0) {
358 throw Exception("Miss",
359 "Data " + name + " not found.");
360 } else if (options.size() > 1) {
361 throw Exception("Ambig",
362 "Data " + name + " is ambiguous. Provide a pass name.");
363 }
364
365 // exactly one option
366 full_name = fullName(options.at(0).name(), options.at(0).pass());
367 type = options.at(0).type();
368 // add into cache
369 known_lookups_[name] = full_name;
370 }
371
372 if (objects_.find(full_name) == objects_.end()) {
373 // final check for input file, never should enter here without one
374 if (not input_file_) {
375 throw Exception("Miss",
376 "Data " + full_name + " was not created by an earlier processor "
377 "and there is not input file to attempt to read it from.");
378 }
379
380 // when setting up the in-memory object, we need to find the tag so we can
381 // 1. get whether the object should be copied to the output file and
382 // 2. set the loaded_ flag to true
383 // we can do a simple linear search since this will only happen once per input file
384 auto tag_it = std::find_if(available_objects_.begin(), available_objects_.end(),
385 [&full_name,this](const EventObjectTag& tag) {
386 return fullName(tag.name(),tag.pass()) == full_name;
387 });
388 tag_it->loaded_ = true;
389
390 // a data set hasn't been created for this data yet
391 // we good, lets create the new data set
392 //
393 // NOTES:
394 // - check if new data is going to be written to output file
395 // or just used during this run
396 // - without any applicable drop/keep rules,
397 // we do save these datasets
398 // - we mark these objects as should_load == false because
399 // they are new and not from an input file
400 auto& obj{objects_[full_name]};
401 obj.data_ = std::make_unique<io::Data<DataType>>(io::constants::EVENT_GROUP+"/"+full_name);
402 obj.should_save_ = tag_it->keep();
403 obj.should_load_ = true;
404 obj.updated_ = false;
405
406 // get this object up to the current entry
407 // loading may throw an H5 error if the shape of the data on disk
408 // cannot be loaded into the input type
409 try {
410 if (not obj.should_save_ or not input_file_->canCopy()) {
411 // only skip the first i_entry_ entries if the input file cannot copy
412 // or the object is not being saved
413 // the objects that are being saved are being mirrored by the input file
414 // if the input file can copy
415 for (std::size_t i{0}; i < i_entry_; i++) input_file_->load_into(*obj.data_);
416 }
417 input_file_->load_into(*obj.data_);
418 } catch (const HighFive::DataSetException&) {
419 throw Exception("BadType",
420 "Data " + full_name + " could not be loaded into "
421 + boost::core::demangle(typeid(DataType).name())
422 + " from the type it was written as " + type);
423 }
424 }
425
426 // type casting, 'bad_cast' thrown if unable
427 try {
428 return objects_[full_name].getDataRef<DataType>().get();
429 } catch (const std::bad_cast&) {
430 throw Exception("BadType",
431 "Data " + full_name + " was initialy loaded with type " + type
432 + " which cannot be casted into "
433 + boost::core::demangle(typeid(DataType).name()));
434 }
